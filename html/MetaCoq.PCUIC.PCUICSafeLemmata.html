Inductive dlexprod {A} {B : A → Type}
          (leA : A → A → Prop) (leB : ∀ x, B x → B x → Prop)
  : sigT B → sigT B → Prop :=
| left_lex : ∀ x x' y y', leA x x' → dlexprod leA leB (x;y) (x';y')
| right_lex : ∀ x y y', leB x y y' → dlexprod leA leB (x;y) (x;y').

        Acc (leB x) y →
        Acc (@dlexprod A B leA leB) (x;y).

Lemma dlexprod_Acc :

Lemma dlexprod_trans :
